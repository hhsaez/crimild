   SET(
        CRIMILD_INCLUDE_DIRECTORIES

   )

   SET(
        CRIMILD_LIBRARY_LINK_LIBRARIES
        ${CRIMILD_LIBRARY_LINK_LIBRARIES}
        ${OPENAL_LIBRARY}
   )

   MESSAGE( "Cannot found package OpenAL" )
ENDIF ()
